{{- /* A row of player heads with a "+N" overflow chip. On hover/focus the chip reveals the hidden players (head + name) - unless the overflow is large, in which case it stays a plain count so the popover never turns into a wall of names. ctx: dict "players" (slice of IGNs, required) "show" (int, heads shown before collapsing; default 4) "popMax" (int, max hidden players the hover roster will list; default 10) */ -}} {{- $players := .players -}} {{- $show := int (.show | default 4) -}} {{- $popMax := int (.popMax | default 10) -}} {{- with $players -}} {{- range first (int (math.Min $show (len .))) . -}} {{ partial "playerhead.html" . }} {{- end -}} {{- with after $show $players -}} +{{ len . }} {{- if le (len .) $popMax -}} {{- range . -}} {{ partial "playerhead.html" . }} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
